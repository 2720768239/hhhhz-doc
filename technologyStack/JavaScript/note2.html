手动实现浅拷贝

js
const shallowClone = (target) => {
  if (typeof target === 'object' && target !== null) {
    let cloneTarget = Array.isArray(target) ? [] : {}

    for (let prop in target) {
      if (target.hasOwnProperty(prop)) {
        cloneTarget[prop] = target[prop]
      }
    }
    return cloneTarget
  } else {
    return target
  }
}

浅拷贝知识创建了一个新的对象，复制了原有对象的基本类型的值。

深拷贝：对于复杂引用数据类型，其在堆内存中完全开辟了一个内存地址，并将原有的对象完全复制过来存放

深拷贝：在堆内存中开辟一块新的内存地址，并将原有的对象完全复制过来存放。

  • JSON.parse(JSON.stringify(obj))

注意：

  1. 不能处理函数、undefined、Symbol。因为 JSON.stringify()无法处理函数、undefined、Symbol
  2. 拷贝 Date 引用类型会变成字符串
  3. 不能拷贝枚举的属性
  4. 不能拷贝对象的原型链
  5. 不能拷贝正则

let obj1 = { a: 1, b: { c: 2 } }
const deepClone = (obj) => {
  let cloneObj = {}
  for (let key in obj) {
    if (typeof obj[key] === 'object') {
      // 如果是对象，就再次调用 deepClone递归
      cloneObj[key] = deepClone(obj[key])
    } else {
      // 基本类型直接复制
      cloneObj[key] = obj[key]
    }
  }
}
